# リリース方法


vi vue.config.js
npm run build  (※)
# スクリプトをサーバにアップロード
scp -P 51022 -rp dist sl-navi.com:/var/www/sl-navi.com/static/
# スクリプトをblobにアップロード
ssh sl-navi
   /usr/local/bin/azcopy login --identity
   /usr/local/bin/azcopy 'sync' --recursive '/var/www/static/' 'https://slnaviblobstrage.blob.core.windows.net/static/'
   sudo service gunicorn restart

※ package.jsonにbuildはvue.config.js、serveはvue.config-serv.jsを使うように記載してある
    "serve": "VUE_CLI_SERVICE_CONFIG_PATH=$PWD/vue.config-serve.js vue-cli-service serve",
    "build": "vue-cli-service build",
    "lint": "vue-cli-service lint"




# sl-navi開発環境作成手順(

## npmのバージョンアップ
npm --version
npm install -g npm

## vue-cliインストール
cd pj-folder
npm uninstall -g @vue/cli
npm   install -g @vue/cli

## sl-naviのテンプレ作成
vue create sl-navi
cd sl-navi
vue add router
npm install vue bootstrap-vue bootstrap
npm install  axios vue-axios
npm install vue-head --save
いったん確認
npm run serve

## gitのファイルをここへ持ってくる(必要なファイルは以下のみ)
cp save/src/* sl-navi/src

## distが相対パスで動くように
# 相対パスのbuild作成(PJフォルダで)
cat > sl-navi/vue.config.js
module.exports = {
  publicPath: './'
}

## build
npm run build
distを好きなところへ持って行く


## fullcalendar
npm install --save @fullcalendar/vue @fullcalendar/daygrid
npm install --save @fullcalendar/timegrid

## イベント登録
npm install --save vue-ctk-date-time-picker
